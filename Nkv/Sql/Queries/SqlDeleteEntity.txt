alter procedure nkv_Delete{0}Entity
(
	@key nvarchar(128),
	@timestamp datetime,
	@ignoreLock tinyint = 0
)
as
begin	
	declare @rowTimestamp datetime;
	declare @lockTimestamp datetime;
	declare @rowCount int;
	declare @ackCode varchar(32) = 'Success'
	declare @version bigint = 0;

	delete from [{0}] where [Key] = @key and [Timestamp] = @timestamp and (@ignoreLock = 1 or [LockTimestamp] is null);
                
	set @rowCount = @@rowcount;
	if @rowCount <> 1
	begin
		select @rowTimestamp = [Timestamp], @lockTimestamp = [LockTimestamp] from [{0}] where [Key] = @key;
		set @ackCode =
		case
			when @rowTimestamp is null then 'KeyNotFound'
			when @rowTimestamp <> @timestamp then 'VersionMismatch'
			when @lockTimestamp is not null then 'EntityLocked'
			else 'Unknown'
		end;
	end

	select @rowCount [RowCount], @ackCode [AckCode], @rowTimestamp [Timestamp], @version [Version];
end
