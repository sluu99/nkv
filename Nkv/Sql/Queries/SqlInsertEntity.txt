alter procedure dbo.nkv_Insert{0}Entity
(
	@key nvarchar(128),
	@value nvarchar(max)
) as
begin
	declare @timestamp datetime = null;
	declare @ackCode varchar(32);
	declare @rowCount int;

	select @timestamp = [Timestamp]
	from dbo.[{0}]
	where [Key] = @key;

	if @timestamp is not null
	begin
		set @ackCode = 'KeyExists';
		set @rowCount = 0;
	end
	else
	begin
		set @timestamp = sysutcdatetime();
	
		insert into dbo.[{0}]([Key], [Value], [Timestamp])
		values(@key, @value, @timestamp);
	
		set @rowCount = @@ROWCOUNT;
		if @rowCount = 1
		begin
			set @ackCode = 'Success';
		end
		else
		begin
			set @ackCode = 'Unknown';
		end
	end

	select @rowCount [RowCount], @timestamp [Timestamp], @ackCode [AckCode];
end
